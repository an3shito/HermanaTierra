CREATE TABLE Tarjeta (
    TarjetaID INT IDENTITY(1,1) PRIMARY KEY,
    No_Hoja INT,
    Responsable VARCHAR(255),
    Articulo VARCHAR(255),
    Especificacion VARCHAR(255),
    Descripcion TEXT,
    Fecha DATE,
    No_Ref VARCHAR(100),
    Rem_Des VARCHAR(100),
    Cant_Entrada INT,
    Precio_U_Ent DECIMAL(10, 2), -- Utilizamos DECIMAL para representar precios con precisi√≥n
    Total_Entrada DECIMAL(10, 2),
    Fecha_Ven DATE,
    Lote VARCHAR(15),
    Cant_Salida INT,
    Precio_Salida DECIMAL(10, 2),
    Ajuste VARCHAR(10),
    Cantidad_Saldo INT,
    Precio_Saldo DECIMAL(10, 2),
    Observacion TEXT
);

CREATE TABLE PrecioHistorico (
    PrecioHistoricoID INT IDENTITY(1,1)  PRIMARY KEY,
    TarjetaID INT,
    FechaInicio DATE,
    FechaFin DATE,
    Precio DECIMAL(10, 2),
    FOREIGN KEY (TarjetaID) REFERENCES Tarjeta(TarjetaID)
);
